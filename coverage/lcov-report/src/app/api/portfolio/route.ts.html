import { NextRequest, NextResponse } from "next/server";
import { fetchBackend, getBackendUserId } from "@/lib/proxy";
 
const BACKEND = process.env.BACKEND_URL ?? "https://trading-app-production-8775.up.railway.app";
 
interface BackendPosition {
  symbol: string;
  totalQuantity: number;
  avgPrice: string | number;
}
 
interface BackendPortfolio {
  totalInvested: string | number;
  positions: BackendPosition[];
}
 
/** Read the first SSE price event from the backend and return the price. */
async function fetchCurrentPrice(symbol: string, cookieHeader: string): Promise<number | null> {
  const ac = new AbortController();
  const timeout = setTimeout(() => ac.abort(), 5000);
 
  try {
    const res = await fetch(
      `${BACKEND}/api/prices?symbol=${encodeURIComponent(symbol)}`,
      { headers: { Cookie: cookieHeader }, signal: ac.signal }
    );
 
    if (!res.ok || !res.body) return null;
 
    const reader = res.body.getReader();
    const dec = new TextDecoder();
    let buf = "";
 
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
 
      buf += dec.decode(value, { stream: true });
      for (const line of buf.split("\n")) {
        if (!line.startsWith("data:")) continue;
        const raw = line.slice(5).trim();
        if (!raw) continue;
        try {
          const msg: { price: number } = JSON.parse(raw);
          if (typeof msg.price === "number") {
            reader.cancel();
            return msg.price;
          }
        } catch {}
      }
      // Keep only the last incomplete line in buf
      const lines = buf.split("\n");
      buf = lines.pop() ?? "";
    }
 
    return null;
  } catch {
    return null;
  } finally {
    clearTimeout(timeout);
  }
}
 
export async function GET(req: NextRequest) {
  const cookieHeader = req.headers.get("cookie") ?? "";
 
  const userId = await getBackendUserId(cookieHeader);
  if (!userId) {
    return NextResponse.json(
      { success: false, error: { code: 401, message: "Unauthorized" } },
      { status: 401 }
    );
  }
 
  const backendRes = await fetchBackend(`/portfolio/${userId}`, {}, cookieHeader);
  if (!backendRes.ok) {
    const body = await backendRes.json();
    return NextResponse.json(body, { status: backendRes.status });
  }
 
  const { data }: { data: BackendPortfolio } = await backendRes.json();
 
  const totalInvested = parseFloat(String(data.totalInvested));
  const rawPositions = data.positions ?? [];
 
  // Fetch live prices for all unique symbols in parallel
  const uniqueSymbols = [...new Set(rawPositions.map((p) => p.symbol))];
  const priceMap = Object.fromEntries(
    await Promise.all(
      uniqueSymbols.map(async (sym) => [sym, await fetchCurrentPrice(sym, cookieHeader)])
    )
  ) as Record<string, number | null>;
 
  const positions = rawPositions.map((p) => {
    const avgCost = parseFloat(String(p.avgPrice));
    const currentPrice = priceMap[p.symbol] ?? avgCost;
    const pnl = parseFloat(((currentPrice - avgCost) * p.totalQuantity).toFixed(2));
    const pnlPercent = avgCost > 0
      ? parseFloat((((currentPrice - avgCost) / avgCost) * 100).toFixed(2))
      : 0;
 
    return {
      symbol: p.symbol,
      quantity: p.totalQuantity,
      avgCost,
      currentPrice,
      pnl,
      pnlPercent,
    };
  });
 
  const totalValue = parseFloat(
    positions.reduce((sum, p) => sum + p.currentPrice * p.quantity, 0).toFixed(2)
  );
  const totalPnl = parseFloat(positions.reduce((sum, p) => sum + p.pnl, 0).toFixed(2));
  const totalPnlPercent =
    totalInvested > 0
      ? parseFloat(((totalPnl / totalInvested) * 100).toFixed(2))
      : 0;
 
  return NextResponse.json({
    success: true,
    data: {
      totalValue,
      totalPnl,
      totalPnlPercent,
      positions,
    },
  });
}
